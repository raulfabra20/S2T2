CONSULTAS RESUMEN

1. SELECT COUNT(*) AS total_alumnos FROM persona p WHERE p.tipo = "alumno";

2. SELECT COUNT(*) AS total_alumnos_nacidos_1999 FROM persona p WHERE YEAR(p.fecha_nacimiento) = 1999;

3. SELECT (d.nombre) AS nombre_departamento, COUNT(pr.id_profesor) AS numero_profesores_por_departamento FROM departamento d JOIN profesor pr ON d.id = pr.id_departamento GROUP BY d.nombre HAVING numero_profesores_por_departamento > 0 ORDER BY numero_profesores_por_departamento DESC;

4. SELECT d.nombre, COUNT(pr.id_profesor) AS numero_profesores FROM profesor pr LEFT JOIN departamento d ON pr.id_departamento = d.id GROUP BY d.nombre;

5. SELECT g.nombre, COUNT(a.id) AS numero_asignaturas FROM grado g LEFT JOIN asignatura a ON g.id = a.id_grado GROUP BY g.nombre ORDER BY numero_asignaturas DESC;

6. SELECT g.nombre, COUNT(a.id) AS numero_asignaturas FROM grado g LEFT JOIN asignatura a ON g.id = a.id_grado GROUP BY g.nombre HAVING numero_asignaturas >= 40;

7. SELECT g.nombre, a.tipo, COUNT(a.creditos) AS numero_creditos FROM grado g LEFT JOIN asignatura a ON g.id = a.id_grado GROUP BY g.nombre;

8. SELECT ce.anyo_inicio, COUNT(asma.id_alumno) AS numero_alumnos_matriculados FROM curso_escolar ce LEFT JOIN alumno_se_matricula_asignatura asma ON ce.id = asma.id_alumno GROUP BY ce.anyo_inicio;

9. SELECT p.id, p.nombre, p.apellido1, p.apellido2, COUNT(a.id) AS numero_asignaturas FROM persona p LEFT JOIN profesor pr ON p.id = pr.id_profesor LEFT JOIN asignatura a ON pr.id_profesor = a.id_profesor GROUP BY pr.id_profesor ORDER BY numero_asignaturas DESC;

10. SELECT * FROM persona p WHERE p.tipo = "alumno" ORDER BY p.fecha_nacimiento DESC LIMIT 1;

11. SELECT p.apellido1, p.apellido2, p.nombre FROM persona p LEFT JOIN profesor pr ON p.id = pr.id_profesor LEFT JOIN departamento d ON pr.id_departamento = d.id LEFT JOIN asignatura a ON pr.id_profesor = a.id_profesor WHERE p.tipo = "profesor" AND a.id IS NULL;
